  border: 1px solid var(--color-border-default);
  border-radius: 8px;
  color: var(--color-text-primary);
  transition: all 150ms ease;
}

.input-field:focus {
  outline: none;
  border-color: var(--color-border-focus);
  box-shadow: 0 0 0 3px rgba(0,212,255,0.1);
}
